    Change comes in two flavors

  • adding things (additive change)
  • removing things (breaking change)
    • Adding things is good
    Removing things is incredibly satisfying 

    TypeError: undefined is not a function

    Only remove after 💯% certain there are no references!

    Breaking changes to JS are extremely rare because browsers that break lose adoption. *

    * But we do have a long history of browsers forcing adoption of _additive features_ by implementing/inventing their own speical browser specific non-standard things..

    Standards are very important because non-standard things tend to go away.

  /* /js/menu-toggle.mjs */
  /* global document */
  var open = document.getElementById('menu-open')
  var close = document.getElementById('menu-close')
  var main = document.querySelector('.main')
  
  export default function MenuToggle () {
    function toggle (e) {
      e.preventDefault()
      main.classList.toggle('slide-menu')
    }

    if (open) {
      open.onclick = toggle
    }

    if (close) {
      close.onclick = toggle
    }
  }
  
  
    
  /* /js/close-button.mjs */
  import {h} from 'https://unpkg.com/preact@8.2.9/dist/preact.esm.js?module'
  import Icon from '/js/icon.mjs'
  import joinClasses from '/js/join-classes.mjs'

  export default function CloseButton(props={}) {
    let noop = e=> {}
    let class = joinClasses('cu-pointer p-2 bg-p0', props.class)
    let onclick = props.onclick || noop
    let attr = {class, onclick}
    let children = h(Icon, {class:'icon', href:'close'})
    return h('button', attributes, children)
  }
